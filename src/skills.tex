\section*{Skills and Competences}
{ % Scoped. Make below config, commands only available for this section

  % Constants
  \newcommand{\skillNaneWidthPercentage}{0.25}
  \newcommand{\expertLevel}{100} % I have no idea what define an expert actually
  \newcommand{\getJobDoneLevel}{40} % Am I too optimistic?

  % Define a command to draw a rectangle which width is calculate to given
  % percentage of minipage width
  \newcommand{\horizontalBox}[1]{%
    \FPeval{\w}{clip(#1 / 100 * (1-\skillNaneWidthPercentage))}
    \colorbox{primaryColor}{\vphantom{0}\hspace{\w\textwidth}}
  }

  % Define a new command to draw a line in skill table.
  % Args:
  % - Skill name: text
  % - Skill level: int, from 0-100 (%)
  \newcommand{\skill}[2]{%
  #1 & \horizontalBox{#2} \\
  }

  % increase margin between rows
  \def\arraystretch{1.5}

  \begin{tabular*}{\linewitdth}{
      @{}>{\raggedleft}p{\skillNaneWidthPercentage\linewidth-2\tabcolsep}@{}
      % without \arraybackslash, \\ won't end the row
      @{}>{\raggedright\arraybackslash}l@{}
    }

    % Draw the expert level indicator
    % Random thoughts: the indicator point to 95% (as per above computation)
    % I could find some trick to force it point to 100%. But I think no one could
    % claim that they understand 100% on any skills, so, 95% is fine.
    & \FPeval{\w}{clip(\expertLevel /100 * (1-\skillNaneWidthPercentage))}%
      \begin{tabular}{@{}>{\raggedleft}p{\w\linewidth}|} %
       \arrayrulecolor{primaryColor}
       \textit{Expert}
      \end{tabular} \\

    % List of skills with level percentage
    \skill{Java}{75}
    \skill{Android}{65}
    \skill{C++}{45}
    \skill{Node.js}{55}
    \skill{Web}{60}

    % Draw the get job done indicator
    & \FPeval{\w}{clip(\getJobDoneLevel /100  * (1-\skillNaneWidthPercentage))}%
      \begin{tabular}{
        @{}p{\w\linewidth + 2\tabcolsep}@{}|l@{}
        }%
        \arrayrulecolor{primaryColor}
        & \textit{Get job done}
      \end{tabular}
  \end{tabular*}
}%
